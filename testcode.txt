function(element, fileName) {
    var ROW_SEL    = '.qa-grid-row';
    var CELL_SEL   = '[data-key]';
    var HDR_SEL    = '[data-column-key]';
    var HDR_LABEL  = '.qa-grid-filter-label';
    var TOT_PAGES  = '.qa-total-grid-pages';
    var BODY_SEL   = '.qa-grid-body.form-scroll';

    function txt(el) { return el ? (el.innerText || el.textContent || '').trim() : ''; }
    function xmlEsc(v) {
        return String(v == null ? '' : v).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/'/g, '&apos;');
    }

    // 1. Setup Persistent Storage on the window so it doesn't vanish
    window.scrapingData = [];
    window.seenIds = new Set();
    
    var tElem = document.querySelector(TOT_PAGES);
    var totalPages = tElem ? parseInt(tElem.textContent, 10) : 1;
    var currentPage = 1;

    // 2. Initialize Headers
    var headers = [];
    var keys = [];
    var firstRow = document.querySelector(ROW_SEL);
    if (firstRow) {
        var cs = firstRow.querySelectorAll(CELL_SEL);
        for (var j = 0; j < cs.length; j++) { keys.push(cs[j].getAttribute('data-key') || ''); }
    }
    var hs = document.querySelectorAll(HDR_SEL);
    for (var h = 0; h < hs.length; h++) {
        var l = hs[h].querySelector(HDR_LABEL) || hs[h];
        headers.push(txt(l));
    }
    window.scrapingData.push(headers);

    // 3. The Extraction Logic
    function scrapeCurrentPage() {
        var currentRows = document.querySelectorAll(ROW_SEL);
        currentRows.forEach(function(row) {
            var rowId = row.getAttribute('data-id') || txt(row);
            if (!window.seenIds.has(rowId)) {
                window.seenIds.add(rowId);
                var vals = [];
                var cells = row.querySelectorAll(CELL_SEL);
                for (var c = 0; c < keys.length; c++) {
                    var el = cells[c] ? (cells[c].querySelector('a,span') || cells[c]) : null;
                    vals.push(txt(el));
                }
                window.scrapingData.push(vals);
            }
        });

        if (currentPage < totalPages) {
            var btn = null;
            var icons = document.querySelectorAll('i.material-icons');
            for (var i = 0; i < icons.length; i++) {
                if (txt(icons[i]) === 'chevron_right') { btn = icons[i].parentNode; break; }
            }

            if (btn) {
                currentPage++;
                btn.click();
                // Wait for the grid body to change, then scrape again
                waitForChanges();
            } else {
                generateFile();
            }
        } else {
            generateFile();
        }
    }

    function waitForChanges() {
        var target = document.querySelector(BODY_SEL);
        var observer = new MutationObserver(function(mutations, obs) {
            obs.disconnect(); // Stop watching
            setTimeout(scrapeCurrentPage, 1000); // Small buffer for rendering
        });
        observer.observe(target, { childList: true, subtree: true });
    }

    function generateFile() {
        var rowsXml = '';
        window.scrapingData.forEach(function(row) {
            rowsXml += '<Row>';
            row.forEach(function(val) {
                rowsXml += '<Cell><Data ss:Type=\'String\'>' + xmlEsc(val) + '</Data></Cell>';
            });
            rowsXml += '</Row>';
        });

        var xml = '<?xml version=\'1.0\'?><Workbook xmlns=\'urn:schemas-microsoft-com:office:spreadsheet\' xmlns:ss=\'urn:schemas-microsoft-com:office:spreadsheet\'><Worksheet ss:Name=\'Data\'><Table>' + rowsXml + '</Table></Worksheet></Workbook>';
        var blob = new Blob([xml], { type: 'application/vnd.ms-excel' });
        var a = document.createElement('a');
        a.href = URL.createObjectURL(blob);
        a.download = fileName;
        document.body.appendChild(a);
        a.click();
        a.remove();
        
        // Clean up window object
        delete window.scrapingData;
        delete window.seenIds;
    }

    // Start the chain
    scrapeCurrentPage();
    return "Scraper running in background. Please wait for the download.";
}
